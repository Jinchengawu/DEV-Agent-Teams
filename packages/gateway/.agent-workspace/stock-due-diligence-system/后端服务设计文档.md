```python
# financial_calculator.py - 核心计算逻辑伪代码

class FinancialCalculator:
    """财务指标计算引擎"""
    
    def calculate_indicators(self, company_code: str, report_date: date) -> dict:
        """计算给定公司和报告期的全部财务指标"""
        
        # 1. 获取原始财务报表数据
        income = self.get_statement(company_code, report_date, 'income')
        balance = self.get_statement(company_code, report_date, 'balance')
        cashflow = self.get_statement(company_code, report_date, 'cashflow')
        
        # 2. 获取上期数据 (用于计算增长率)
        prev_income = self.get_prev_statement(company_code, report_date, 'income')
        
        # 3. 计算盈利能力
        profitability = {
            'gross_margin': income['gross_profit'] / income['revenue'],
            'net_margin': income['net_income'] / income['revenue'],
            'roe': income['net_income'] / balance['total_equity'],
            'roa': income['net_income'] / balance['total_assets'],
            'roic': self._calc_roic(income, balance),
        }
        
        # 4. 计算成长能力
        growth = {
            'revenue_growth_yoy': (income['revenue'] / prev_income['revenue'] - 1),
            'net_income_growth_yoy': (income['net_income'] / prev_income['net_income'] - 1),
            'revenue_growth_cagr_5y': self._calc_cagr(company_code, 'revenue', 5),
        }
        
        # 5. 计算营运能力
        efficiency = {
            'asset_turnover': income['revenue'] / balance['total_assets'],
            'inventory_turnover': income['cost_of_revenue'] / balance['inventory'],
            'receivable_turnover': income['revenue'] / balance['accounts_receivable'],
        }
        
        # 6. 计算偿债能力
        leverage = {
            'debt_to_asset': balance['total_liabilities'] / balance['total_assets'],
            'current_ratio': balance['current_assets'] / balance['current_liabilities'],
            'interest_bearing_debt_ratio': balance['interest_bearing_debt'] / balance['total_assets'],
        }
        
        # 7. 计算现金质量
        ocf = cashflow['operating_cashflow']
        capex = abs(cashflow['capital_expenditure'])
        cashflow_quality = {
            'ocf_to_net_income': ocf / income['net_income'],
            'fcf': ocf - capex,
            'capex_to_depreciation': capex / income['depreciation_amortization'],
        }
        
        # 8. 计算杜邦分析
        dupont = {
            'roe': profitability['roe'],
            'net_margin': profitability['net_margin'],
            'asset_turnover': efficiency['asset_turnover'],
            'equity_multiplier': balance['total_assets'] / balance['total_equity'],
        }
        
        # 9. 计算综合评分
        overall_score = self._calculate_overall_score(
            profitability, growth, efficiency, leverage, cashflow_quality
        )
        
        return {
            'profitability': profitability,
            'growth': growth,
            'efficiency': efficiency,
            'leverage': leverage,
            'cashflow_quality': cashflow_quality,
            'dupont': dupont,
            'overall_score': overall_score,
        }
    
    def _calculate_overall_score(self, *categories) -> float:
        """
        综合评分算法:
        - 盈利能力权重: 30%
        - 成长能力权重: 20%
        - 营运能力权重: 15%
        - 偿债能力权重: 15%
        - 现金质量权重: 20%
        
        每个维度的评分为该维度各指标的行业百分位数加权平均
        """
        weights = [0.30, 0.20, 0.15, 0.15, 0.20]
        scores = []
        for category in categories:
            # 计算该维度的行业百分位评分
            dimension_score = self._calc_dimension_percentile_score(category)
            scores.append(dimension_score)
        
        return sum(s * w for s, w in zip(scores, weights))
```

```python
# dcf_model.py - DCF 估值模型

class DCFModel:
    """DCF 现金流折现模型"""
    
    def calculate(
        self,
        # 基础数据
        latest_revenue: float,
        latest_fcf: float,
        total_shares: float,
        current_price: float,
        net_debt: float,
        
        # 假设参数
        stage1_years: int = 5,
        revenue_growth_rates: list[float] = None,
        net_margin: float = 0.50,
        capex_ratio: float = 0.05,
        working_capital_change: float = 0.02,
        terminal_growth_rate: float = 0.03,
        wacc: float = 0.09,
    ) -> dict:
        """执行 DCF 估值计算"""
        
        projected_fcf = []
        revenue = latest_revenue
        
        # 阶段一: 高速增长期
        for year in range(stage1_years):
            growth = revenue_growth_rates[year] if revenue_growth_rates else 0.10
            revenue = revenue * (1 + growth)
            fcf = revenue * (net_margin - capex_ratio - working_capital_change)
            
            # 折现
            discount_factor = 1 / ((1 + wacc) ** (year + 1))
            pv_fcf = fcf * discount_factor
            
            projected_fcf.append({
                'year': 2024 + year,
                'revenue': revenue,
                'fcf': fcf,
                'discount_factor': discount_factor,
                'pv_fcf': pv_fcf,
            })
        
        # 阶段二: 永续增长期 (终值)
        last_fcf = projected_fcf[-1]['fcf']
        terminal_value = last_fcf * (1 + terminal_growth_rate) / (wacc - terminal_growth_rate)
        pv_terminal = terminal_value / ((1 + wacc) ** stage1_years)
        
        # 企业价值
        pv_stage1 = sum(item['pv_fcf'] for item in projected_fcf)
        enterprise_value = pv_stage1 + pv_terminal
        
        # 股权价值
        equity_value = enterprise_value - net_debt
        value_per_share = equity_value / total_shares
        margin_of_safety = (value_per_share - current_price) / current_price
        
        # 敏感性分析
        sensitivity = self._calc_sensitivity(
            last_fcf, stage1_years, net_debt, total_shares,
            current_price, terminal_growth_rate, wacc
        )
        
        return {
            'enterprise_value': enterprise_value,
            'net_debt': net_debt,
            'equity_value': equity_value,
            'total_shares': total_shares,
            'intrinsic_value_per_share': value_per_share,
            'current_price': current_price,
            'margin_of_safety': margin_of_safety,
            'projected_fcf': projected_fcf,
            'terminal_value': terminal_value,
            'pv_terminal': pv_terminal,
            'sensitivity_table': sensitivity,
        }
    
    def _calc_sensitivity(self, ...):
        """计算 WACC vs 永续增速的敏感性矩阵"""
        wacc_range = [wacc - 0.02, wacc - 0.01, wacc, wacc + 0.01, wacc + 0.02]
        g_range = [g - 0.01, g - 0.005, g, g + 0.005, g + 0.01]
        
        matrix = []
        for w in wacc_range:
            row = []
            for g in g_range:
                if w <= g:
                    row.append(None)  # WACC 必须大于 g
                else:
                    tv = last_fcf * (1 + g) / (w - g)
                    pv_tv = tv / ((1 + w) ** stage1_years)
                    ev = pv_stage1_fixed + pv_tv  # 用固定的 stage1 PV
                    ev_per_share = (ev - net_debt) / total_shares
                    row.append(round(ev_per_share))
            matrix.append(row)
        
        return {
            'wacc_values': wacc_range,
            'terminal_growth_values': g_range,
            'values_per_share': matrix,
        }
```

```python
# risk_detector.py

class RiskDetector:
    """财务风险检测引擎"""
    
    def assess(self, company_code: str) -> dict:
        """综合风险评估"""
        
        categories = [
            self._assess_earnings_quality(company_code),
            self._assess_leverage_risk(company_code),
            self._assess_governance_risk(company_code),
            self._assess_fraud_signals(company_code),
        ]
        
        # 综合评分 (0-100, 越低越安全)
        overall_score = sum(c['score'] * c['weight'] for c in categories)
        overall_level = self._score_to_level(overall_score)
        
        return {
            'overall_score': overall_score,
            'overall_level': overall_level,
            'categories': categories,
            'beneish_m_score': self._calc_beneish_m_score(company_code),
            'altman_z_score': self._calc_altman_z_score(company_code),
        }
    
    def _calc_beneish_m_score(self, company_code: str) -> dict:
        """
        Beneish M-Score 财务舞弊检测模型
        M-Score = -4.84 + 0.92×DSRI + 0.528×GMI + 0.404×AQI + 0.892×SGI
                   + 0.115×DEPI - 0.172×SGAI + 4.679×TATA - 0.327×LVGI
        
        M > -1.78: 高舞弊概率
        M ≤ -1.78: 低舞弊概率
        """
        # DSRI: 应收账款天数指数
        # GMI: 毛利率指数
        # AQI: 资产质量指数
        # SGI: 营收增长指数
        # DEPI: 折旧率指数
        # SGAI: 销管费用指数
        # LVGI: 杠杆指数
        # TATA: 总应计项目/总资产
        
        dsri = self._calc_dsri(company_code)
        gmi = self._calc_gmi(company_code)
        aqi = self._calc_aqi(company_code)
        sgi = self._calc_sgi(company_code)
        depi = self._calc_depi(company_code)
        sgai = self._calc_sgai(company_code)
        lvgi = self._calc_lvgi(company_code)
        tata = self._calc_tata(company_code)
        
        m_score = (
            -4.84
            + 0.920 * dsri
            + 0.528 * gmi
            + 0.404 * aqi
            + 0.892 * sgi
            + 0.115 * depi
            - 0.172 * sgai
            + 4.679 * tata
            - 0.327 * lvgi
        )
        
        return {
            'score': round(m_score, 2),
            'threshold': -1.78,
            'conclusion': '低舞弊概率' if m_score <= -1.78 else '高舞弊概率',
            'components': {
                'dsri': round(dsri, 3),
                'gmi': round(gmi, 3),
                'aqi': round(aqi, 3),
                'sgi': round(sgi, 3),
                'depi': round(depi, 3),
                'sgai': round(sgai, 3),
                'lvgi': round(lvgi, 3),
                'tata': round(tata, 3),
            }
        }
    
    def _calc_altman_z_score(self, company_code: str) -> dict:
        """
        Altman Z-Score 破产风险模型
        Z = 1.2×X1 + 1.4×X2 + 3.3×X3 + 0.6×X4 + 1.0×X5
        
        Z > 2.99: 安全区
        1.81 < Z ≤ 2.99: 灰色区
        Z ≤ 1.81: 危险区
        """
        # X1 = 营运资本/总资产
        # X2 = 留存收益/总资产
        # X3 = EBIT/总资产
        # X4 = 股权市值/总负债
        # X5 = 营收/总资产
        
        x1 = self._calc_x1(company_code)
        x2 = self._calc_x2(company_code)
        x3 = self._calc_x3(company_code)
        x4 = self._calc_x4(company_code)
        x5 = self._calc_x5(company_code)
        
        z = 1.2*x1 + 1.4*x2 + 3.3*x3 + 0.6*x4 + 1.0*x5
        
        if z > 2.99:
            zone = '安全区'
        elif z > 1.81:
            zone = '灰色区'
        else:
            zone = '危险区'
        
        return {
            'score': round(z, 2),
            'zone': zone,
            'threshold': {'safe': 2.99, 'grey': 1.81},
            'components': {
                'x1': round(x1, 4), 'x2': round(x2, 4),
                'x3': round(x3, 4), 'x4': round(x4, 4),
                'x5': round(x5, 4),
            }
        }
```

```python
# ai_chat_service.py

from langchain.chat_models import ChatOpenAI
from langchain.embeddings import OpenAIEmbeddings
from langchain.vectorstores import Chroma
from langchain.chains import RetrievalQA

class AIChatService:
    """AI 智能问答服务"""
    
    def __init__(self):
        self.llm = ChatOpenAI(model="gpt-4-turbo", temperature=0.1)
        self.embeddings = OpenAIEmbeddings()
        self.vectorstore = Chroma(
            embedding_function=self.embeddings,
            persist_directory="./chroma_db"
        )
    
    async def chat(
        self,
        company_code: str,
        question: str,
        context: list = None,
    ) -> AsyncGenerator[str, None]:
        """流式回答用户关于公司基本面的问题"""
        
        # 1. 获取公司相关数据作为上下文
        company_data = await self._fetch_company_context(company_code)
        
        # 2. 检索相关知识
        relevant_docs = self.vectorstore.similarity_search(
            question, 
            filter={"company_code": company_code},
            k=5
        )
        
        # 3. 构建 Prompt
        system_prompt = f"""你是一个专业的股票基本面分析师。
        
请基于以下数据回答用户的问题。要求：
1. 回答要准确、专业、有深度
2. 引用具体数据支撑观点
3. 如有需要，可以用表格或列表展示数据
4. 给出客观的分析结论

公司数据:
{company_data}

相关历史分析:
{self._format_docs(relevant_docs)}
"""
        
        # 4. 流式生成
        messages = [
            SystemMessage(content=system_prompt),
            *([HumanMessage(content=c) for c in (context or [])]),
            HumanMessage(content=question),
        ]
        
        async for chunk in self.llm.astream(messages):
            yield json.dumps({
                "type": "text",
                "content": chunk.content
            })
        
        yield json.dumps({
            "type": "reference",
            "sources": ["financial_statements", "financial_indicators"]
        })
        yield json.dumps({"type": "done"})
    
    async def _fetch_company_context(self, code: str) -> str:
        """获取公司数据作为 LLM 上下文"""
        company = await self.company_repo.get_by_code(code)
        indicators = await self.financial_repo.get_latest_indicators(code)
        revenue_structure = await self.business_repo.get_revenue_structure(code)
        risk = await self.risk_repo.get_latest_assessment(code)
        
        return f"""
公司: {company.name} ({company.code}.{company.market})
行业: {company.industry}
主营业务: {company.main_business}

最新财务指标:
- ROE: {indicators.roe:.1%}
- 毛利率: {indicators.gross_margin:.1%}
- 净利率: {indicators.net_margin:.1%}
- 资产负债率: {indicators.debt_to_asset:.1%}
- 营收增速(YoY): {indicators.revenue_growth_yoy:.1%}
- PE-TTM: {company.pe_ttm}x
- PB: {company.pb}x

营收结构:
{json.dumps(revenue_structure, ensure_ascii=False, indent=2)}

风险评分: {risk.overall_score}/100 ({risk.overall_level})
"""
```

## 三、缓存策略

### 3.1 Redis 缓存

| 缓存 Key | 内容 | TTL | 失效策略 |
|----------|------|-----|---------|
| `company:{code}:basic` | 公司基本信息 | 1h | 主动失效 |
| `company:{code}:financial:{year}:{type}` | 财务报表 | 24h | 新数据入库时失效 |
| `company:{code}:indicators:{year}` | 财务指标 | 24h | 同上 |
| `company:{code}:valuation` | 估值数据 | 1h | 定时刷新 |
| `company:{code}:risk` | 风险评估 | 6h | 定时刷新 |
| `industry:{code}:companies` | 行业内公司列表 | 2h | 定时刷新 |
| `search:{keyword}` | 搜索结果 | 5min | 短期缓存 |
| `user:{id}:watchlist` | 用户自选股 | 30min | 写操作时失效 |

### 3.2 缓存预热

系统启动时和每日凌晨预热以下数据：
- 沪深300成分股的公司基本信息
- 最近一个报告期的财务指标
- 最新风险评估结果
